#include <bits/stdc++.h>

using namespace std;

int main()
{
    int M;cin >> M;
    int n;cin >> n;
    vector<int> ljudi(n);
    for(int i = 0;i < n;i++){
        cin >> ljudi[i];
    }
    int i = 0;
    int j = n - 1;
    int brodici = 0;
    while(i <= j){
        if(ljudi[i] + ljudi[j] <= M){
            i++;
        }
        j--;
        brodici++;
    }
    cout << brodici << '\n';
    return 0;
}
